 * estranho), convencionalmente o nome de arquivo deveria estar em

 *
 * Fez uso de [RxJS](http://reactivex.io/rxjs/) não deixando nenhum `subscribe` solto aparente sem precisar.
 *
 * Faltou limpar os `console.log`, geralmente são usados apenas enquanto
 * desenvolvendo em ambiente local, não costumamos subir para o repositório
 * principal sem motivo.
 */
@NgModule({
  imports: [
    CommonModule,
    FormsModule,
  ],
  declarations: [SignInFormComponent, RootComponent, SignUpFormComponent]
})
export class AuthModule { }
