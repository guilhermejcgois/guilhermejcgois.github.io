import { Component, OnInit, Input } from '@angular/core';

import { Item, ItemList } from '../models/item';
import { FbDataService } from '../services/fb-data.service';

@Component({
  selector: 'app-list-options',
  templateUrl: './list-options.component.html',
  styleUrls: ['./list-options.component.css']
})

export class ListOptionsComponent implements OnInit {

  @Input('listOfType') listOfType: ItemList;
  newItem: Item = {
    name: '',
    alreadyInCart: false
  };
  removeCheckedListHint = 'Para remover os items marcardos dê um clique duplo';
  removeListHint = 'Para remover a lista compleata dê um clique duplo';
  constructor(private fbDataService: FbDataService) { }

  ngOnInit() {
  }

  addItem() {
    if (this.newItem.name != '') {
      this.fbDataService.addItem(this.listOfType.type, this.newItem);
      this.newItem.name = '';
    }
  }

  removeChecked() {
    console.log(this.listOfType);
    const tmpList = this.listOfType;
    console.log(tmpList);
    tmpList.items.forEach(item => {
      if (item.alreadyInCart == true) {
        console.log(item);
        this.fbDataService.removeItem(this.listOfType.type, item);
      }
    })
  }

  removeList() {
    this.fbDataService.removeList(this.listOfType.type);
  }

}
