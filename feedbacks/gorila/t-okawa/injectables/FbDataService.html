import { Injectable } from '@angular/core';

import { AngularFireDatabase } from 'angularfire2/database';

import { Observable } from 'rxjs/Observable';
import { map }        from 'rxjs/operators';

import { AuthService } from './auth.service';

import { Item, ItemList } from '../models/item';

@Injectable({
  providedIn: 'root'
})

export class FbDataService {

  constructor(private db: AngularFireDatabase,
              private authService: AuthService) {
  }

	get userListRef(): string {
		return `users/${this.authService.currentUserId}/my-list`;
	}

  /**
   * ** [FEEDBACK] ** Não sei como esse `snapshotChanges` funciona, mas é bom
   * tomar cuidado para o callback do `subscribe` não continuar rodando depois
   * desse componente ser destruído ou caso `getItemsList` seja chamado
   * novamente, para evitar isso é possível fazer o `unsubscribe` no `ngOnDestroy`
   * e a cada nova chamada ao método (se Subject != null, por exemplo), mas para
   * isso você precisa de uma referência ao `Subject` que `snapshotChanges` retorna.
   *
   * ** [FEEDBACK] ** Outra possibilidade é trabalhar como
   * [pipeables operators](https://github.com/ReactiveX/rxjs/blob/master/doc/pipeable-operators.md).
   *
   * ** [FEEDBACK] ** Idem para `getItemsForType`
   */
  getItemsList(): ItemList[] {
    let itemsListArray = new Array<ItemList>();
    this.db.list(this.userListRef).snapshotChanges(['child_added', 'child_removed'])
           .subscribe(userList => {
      this.clearArray(itemsListArray);
      userList.forEach((userType, index) => {
        itemsListArray[index] = this.getItemsForType(userType.key);
      });
    });
    return itemsListArray;
  }

  getItemsForType(itemType): ItemList {
    const ref = `${this.userListRef}/${itemType}`;
    let itemArray = new Array<Item>();
    this.db.list(ref).snapshotChanges().subscribe(itemsOfType => {
      this.clearArray(itemArray);
      itemsOfType.forEach((itemOfType, index) => {
        itemArray.push(itemOfType.payload.val() as Item);
      });
    });

    return { type: `${itemType}`, items: (itemArray as Item[]), showOptions: false };
  }

  clearArray(array) {
    while (array.length) {
      array.pop();
    }
  }

  /**
   * ** [FEEDBACK] ** Um método privado com parâmetros opcionais para retornar
   * `ref`, já que possui o mesmo formato usado em 4 lugares diferentes, mesmo
   * que apenas nesse arquivo.
   */
  addItem(type, item) {
    const ref = `${this.userListRef}/${type}/${item.name}`;
    this.db.object(ref).update(item);
  }

  updateItem(type, item) {
    const ref = `${this.userListRef}/${type}/${item.name}`;
    this.db.object(ref).update(item);
  }

  removeItem(type, item) {
    const ref = `${this.userListRef}/${type}/${item.name}`;
    this.db.object(ref).remove();
  }

  removeList(type) {
    const ref = `${this.userListRef}/${type}`;
    this.db.object(ref).remove();
  }
}
